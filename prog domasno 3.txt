#include <bits/stdc++.h>
using namespace std;
struct CBHT {
    vector<list<string>> table;
    CBHT(int n) {
        table.resize(n);
    }
    int h1(const string &w) {
        return w.length() % table.size();
    }
    int h2(const string &w) {
        int s = 0;
        for (char c : w) s += c;
        return s % table.size();
    }
    int h3(const string &w) {
        return (w[0] - 'a') % table.size();
    }
    int h4(const string &w) {
        if (w.length() < 2) return h3(w);
        int v = (w[0]-'a')*26 + (w[1]-'a');
        return v % table.size();
    }
    void insert(const string &w, int mode) {
        int idx;
        if (mode == 1) idx = h1(w);
        else if (mode == 2) idx = h2(w);
        else if (mode == 3) idx = h3(w);
        else idx = h4(w);

        table[idx].push_back(w);
    }
    bool contains(const string &w, int mode) {
        int idx;
        if (mode == 1) idx = h1(w);
        else if (mode == 2) idx = h2(w);
        else if (mode == 3) idx = h3(w);
        else idx = h4(w);

        for (auto &x : table[idx])
            if (x == w) return true;
        return false;
    }
    void print() {
        for (int i=0; i<table.size(); i++) {
            cout << i << ": ";
            for (auto &w : table[i]) cout << w << " ";
            cout << endl;
        }
    }
};
int main() {
    CBHT hash(1031);  
    string words[] = {"the","and","of","to","in","you","that","it","is","for"};
    for (string w : words) hash.insert(w, 4);

    hash.print();

    return 0;
}
